Required Supabase Tables
Run this SQL in your Supabase SQL Editor. Make sure to disable RLS on both tables so the ESP32 can insert/update with the anon key:
-- bin_logs: ESP32 writes sensor data here CREATE TABLE IF NOT EXISTS bin_logs ( id BIGSERIAL PRIMARY KEY, device_id TEXT DEFAULT 'smartbin-01', bio_level INT DEFAULT 0, nb_level INT DEFAULT 0, bio_distance_mm INT, nb_distance_mm INT, bio_servo_open BOOLEAN DEFAULT false, nb_servo_open BOOLEAN DEFAULT false, created_at TIMESTAMPTZ DEFAULT NOW() ); -- bin_commands: Dashboard writes, ESP32 polls & executes CREATE TABLE IF NOT EXISTS bin_commands ( id BIGSERIAL PRIMARY KEY, device_id TEXT DEFAULT 'smartbin-01', command TEXT NOT NULL, status TEXT DEFAULT 'pending', created_at TIMESTAMPTZ DEFAULT NOW(), executed_at TIMESTAMPTZ ); -- IMPORTANT: Disable RLS so anon key can read/write ALTER TABLE bin_logs DISABLE ROW LEVEL SECURITY; ALTER TABLE bin_commands DISABLE ROW LEVEL SECURITY; -- Enable realtime ALTER TABLE bin_logs REPLICA IDENTITY FULL; ALTER TABLE bin_commands REPLICA IDENTITY FULL;
