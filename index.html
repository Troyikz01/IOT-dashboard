How Automatic Notifications Work
🗑
1 · Bin Fills Up
VL53L0X reads ≤90mm (100% full) or 75% threshold crossed. Buzzer fires fullBeep() for 2s.
📡
2 · Dashboard Fires Request
fetch() POST sent to GAS_SCRIPT_URL. ESP32 also sends independently via HTTPClient to same URL.
📧
3 · GAS Sends Email
Google Apps Script receives payload and calls GmailApp.sendEmail() to all configured recipients.

// SmartBin IoT — Google Apps Script Email Relay function doPost(e) { try { var data = JSON.parse(e.postData.contents); var recipients = data.recipients || ""; var subject = data.subject || "SmartBin Alert"; var body = data.body || ""; if (!recipients) return jsonResp({ok:false,msg:"No recipients"}); GmailApp.sendEmail(recipients, subject, body, { htmlBody: data.htmlBody || body, name: "SmartBin IoT System" }); return jsonResp({ok:true,msg:"Email sent"}); } catch(err) { return jsonResp({ok:false,msg:err.toString()}); } } function doGet(e) { return jsonResp({ok:true,msg:"SmartBin GAS relay active"}); } function jsonResp(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }

// Add to SmartBin ESP32 firmware #include <HTTPClient.h> #include <ArduinoJson.h> const char* GAS_URL = "https://script.google.com/macros/s/YOUR_SCRIPT_ID/exec"; const char* RECIPIENTS = "admin@nbsc.edu.ph,janitor@nbsc.edu.ph"; // Call after fullBeep() or threshold crossed bool sendGASAlert(const char* binType, int pct) { if (WiFi.status() != WL_CONNECTED) return false; HTTPClient http; http.begin(GAS_URL); http.addHeader("Content-Type", "application/json"); StaticJsonDocument<512> doc; doc["recipients"] = RECIPIENTS; doc["subject"] = String(binType) + " bin " + pct + "% - SmartBin IoT"; doc["body"] = String(binType) + " bin reached " + pct + "%. Please empty it."; String body; serializeJson(doc, body); int code = http.POST(body); http.end(); return (code == 200); }

ESP32 Code — Insert into bin_logs
Add this to your firmware to log sensor readings to Supabase after every loop.
// SmartBin ESP32 → Supabase bin_logs #include <WiFi.h> #include <HTTPClient.h> #include <ArduinoJson.h> const char* SUPABASE_URL = "https://wjkvrwwpzngwfrxhcrlt.supabase.co"; const char* SUPABASE_KEY = "YOUR_ANON_KEY_HERE"; // Convert mm distance to percentage (matches getStatus() logic) int mmToPercent(int mm) { if (mm <= 90) return 100; if (mm <= 130) return 75; if (mm <= 170) return 50; if (mm <= 210) return 25; return 0; } // Call this in loop() or on threshold change bool logToSupabase(int d1, int d2, bool bioOpen, bool nbOpen, String bioSt, String nbSt, String eventType = "reading", String alertBin = "") { if (WiFi.status() != WL_CONNECTED) return false; HTTPClient http; http.begin(String(SUPABASE_URL) + "/rest/v1/bin_logs"); http.addHeader("apikey", SUPABASE_KEY); http.addHeader("Authorization", "Bearer " + String(SUPABASE_KEY)); http.addHeader("Content-Type", "application/json"); http.addHeader("Prefer", "return=minimal"); StaticJsonDocument<512> doc; doc["bio_level"] = mmToPercent(d1); doc["nb_level"] = mmToPercent(d2); doc["bio_distance_mm"] = d1; doc["nb_distance_mm"] = d2; doc["bio_status"] = bioSt; doc["nb_status"] = nbSt; doc["bio_servo_open"] = bioOpen; doc["nb_servo_open"] = nbOpen; doc["event_type"] = eventType; if (alertBin.length() > 0) doc["alert_bin"] = alertBin; String body; serializeJson(doc, body); int code = http.POST(body); http.end(); return (code == 201); // Supabase returns 201 on insert } // In loop(), log every ~5 seconds: // static unsigned long lastLog = 0; // if (millis() - lastLog > 5000) { // lastLog = millis(); // logToSupabase(d1, d2, bioServoOpen, nonBioServoOpen, bioStatus, nonBioStatus); // }
