Required Supabase SQL
Run once in your Supabase SQL Editor. The DISABLE ROW LEVEL SECURITY lines are required — without them the ESP32 anon key cannot read commands.
CREATE TABLE IF NOT EXISTS bin_logs ( id BIGSERIAL PRIMARY KEY, device_id TEXT DEFAULT 'smartbin-01', bio_level INT DEFAULT 0, nb_level INT DEFAULT 0, bio_distance_mm INT, nb_distance_mm INT, bio_servo_open BOOLEAN DEFAULT false, nb_servo_open BOOLEAN DEFAULT false, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE IF NOT EXISTS bin_commands ( id BIGSERIAL PRIMARY KEY, device_id TEXT DEFAULT 'smartbin-01', command TEXT NOT NULL, status TEXT DEFAULT 'pending', created_at TIMESTAMPTZ DEFAULT NOW(), executed_at TIMESTAMPTZ ); -- REQUIRED: allow anon key to read/write ALTER TABLE bin_logs DISABLE ROW LEVEL SECURITY; ALTER TABLE bin_commands DISABLE ROW LEVEL SECURITY; -- Enable realtime push ALTER TABLE bin_logs REPLICA IDENTITY FULL; ALTER TABLE bin_commands REPLICA IDENTITY FULL;
