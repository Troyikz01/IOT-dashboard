Google Apps Script Email Alerts
Automated email notifications via GAS relay — no OAuth required
How Automatic Notifications Work
🗑
1 · Bin Fills Up
ESP32 VL53L0X sensor detects distance = 0 mm (100% full) or dashboard simulates the threshold.
📡
2 · Dashboard Fires Request
A fetch() POST is automatically sent to your GAS Web App URL with bin type, level, and timestamp.
📧
3 · GAS Sends Email
Google Apps Script receives the payload and calls GmailApp.sendEmail() to all configured recipients.

// SmartBin IoT — Google Apps Script Email Relay // Paste into script.google.com → Deploy as Web App // Execute as: Me | Access: Anyone function doPost(e) { try { var data = JSON.parse(e.postData.contents); var recipients = data.recipients || ""; var subject = data.subject || "SmartBin Alert"; var body = data.body || ""; if (!recipients) { return jsonResp({ok:false,msg:"No recipients"}); } GmailApp.sendEmail(recipients, subject, body, { htmlBody: data.htmlBody || body, name: "SmartBin IoT System" }); return jsonResp({ok:true,msg:"Email sent"}); } catch(err) { return jsonResp({ok:false,msg:err.toString()}); } } function doGet(e) { return jsonResp({ok:true,msg:"SmartBin GAS relay active"}); } function jsonResp(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }

ESP32 Code — Send Alert Directly from Hardware
Use this in your Arduino / PlatformIO sketch so the ESP32 itself can trigger emails without the dashboard.
// SmartBin ESP32 → Google Apps Script Email Relay // Automatically fires when bin level reaches threshold #include <WiFi.h> #include <HTTPClient.h> #include <ArduinoJson.h> const char* GAS_URL = "https://script.google.com/macros/s/YOUR_SCRIPT_ID/exec"; const char* RECIPIENTS = "admin@nbsc.edu.ph,janitor@nbsc.edu.ph"; bool sendGASAlert(const char* binType, int level) { if (WiFi.status() != WL_CONNECTED) return false; HTTPClient http; http.begin(GAS_URL); http.addHeader("Content-Type", "application/json"); // Build JSON payload StaticJsonDocument<512> doc; doc["recipients"] = RECIPIENTS; doc["subject"] = String(binType) + " bin FULL — SmartBin IoT"; doc["body"] = String(binType) + " bin reached 100%. Please empty it."; String body; serializeJson(doc, body); int code = http.POST(body); http.end(); return (code == 200); } // In your main loop, call this when bin is detected full: void checkAndNotify() { if (bioLevel >= 100 && !bioAlertSent) { sendGASAlert("Biodegradable", bioLevel); bioAlertSent = true; } if (nbLevel >= 100 && !nbAlertSent) { sendGASAlert("Non-Biodegradable", nbLevel); nbAlertSent = true; } }
